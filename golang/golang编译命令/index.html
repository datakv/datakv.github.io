golang 编译命令

go build

go build -gcflags "-l -m"    禁止函数内联,输出编译器优化策略

go build -gcflags "-l -m"   输出编译器优化策略

go build -gcflags "-N -l"   禁止函数内联，禁止代码优化

go tool objdump -s "main.TTT" function.txt 查看汇编代码
