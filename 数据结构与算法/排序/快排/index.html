func QuickPartition(a []int, l, r int) int {
    i := l - 1
    m := r - 1
    for j := l; j < r-1; j++ {
        if a[j] < a[m] {
            i++
            e(a, i, j)
        }
    }
    i++
    e(a, i, r-1)

    return i
}

func QuickInner(a []int, begin, end int) {
    if begin+1 >= end {
        return
    }
    p := QuickPartition(a, begin, end)
    QuickInner(a, begin, p)
    QuickInner(a, p+1, end)
}

func Quick(a []int) { //不稳定
    length := len(a)
    if length < 2 {
        return
    }
    QuickInner(a, 0, length)
}
